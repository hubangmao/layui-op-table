全部扩展参数一览表
对照查看
			layui.opTable.render({
          elem: '#test'
          , id: '#test'
          , url: 'test.json'
          , toolbar: '#barDemo'
          , cols: [[
            {field: 'id', title: 'ID', sort: true}
            , {field: 'username', title: '用户名', edit: true}
            , {field: 'city', title: '城市', edit: true}
            , {title: '操作', toolbar: '#barDemo', width: 150}
          ]]

          //注：展开的列Items
          , openCols: [
            {
              field: 'sign', title: '签名', templet: function (item) {
                // 重新定义排版
                return ""
              }
            }
            , {
              field: 'wealth', title: '财富', onDraw: function (it, item) {
                // 使用默认风格排版 【左边title+右边值+下划线】
                return item['wealth'];
              }
            }
            , {field: 'experience', title: '积分'}
            , {field: 'classify', title: '职业'}
          ]

          //注：监听排序 {opTable}
          , sort: function (obj) {
            console.log(obj)
          }
          //注：0：同时只展开一项 1：一直展开不关闭之前  def：0 {opTable}
          , openType: 0

        });
